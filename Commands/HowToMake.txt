//そのまま書く
#include "Command.cpp"

//名前をCom_*****に変更する
class Com_hello : public Command {
private:
	//コマンドの処理中に使うデータを宣言しておく
	//prepare()内で初期化する
	User* user;

public:
	//Com_hello()のところを各コマンド名に変更する
	//下3行は全く同じ
	Com_hello();
	void makeKeywordList() override;
	void prepare(User* user) override;
	void execute() override;
};  //最後のセミコロンを忘れないようにする

//コンストラクタ
//Com_helloから各コマンド名に変更する
//中身は基本そのままでOK
inline Com_hello::Com_hello() {
	makeKeywordList();
}

//キーワードを設定する
//keywordsという配列に順番に格納しているので、
//keywordsの[]の中身の数字は0から順番に入れていく
//最大20個
inline void Com_hello::makeKeywordList() {
	plug(keywords[0], L"hello");
	plug(keywords[1], L"hi");
}

//コマンドの処理中につかうデータを初期化する
//ユーザーデータなどを全く使用しない場合は空の関数でOK
//ユーザーデータ以外のデータを取得したい場合は要相談
inline void Com_hello::prepare(User* user) {
	this->user = user;
}

//実際に実行されるコマンド
//今まで使ってきたmain()のようなイメージ
inline void Com_hello::execute() {
	printf(L"hi ");
	printf(user->name);
	printf(L"\r\n");
}
